#+begin_src agda2
GROUP : (i : Level) → Precategory (lsuc i) i
ob (GROUP i) = Group i
hom (GROUP i) = Group-hom {i} {i}
hom-set (GROUP i) G H = Group-hom-is-set
Id (GROUP i) = Idᵍ
comp (GROUP i) = ghom-comp
IdL (GROUP i) = ghom-lId
IdR (GROUP i) = ghom-rId
Assoc (GROUP i) = ghom-assoc
