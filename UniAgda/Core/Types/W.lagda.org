#+begin_src agda2 :tangle yes
data W {i j : Level} (A : Type i) (B : A → Type j) : Type (i ⊔ j) where
  sup : (a : A) → (B a → W A B) → W A B

syntax W A (λ x → B) = W[ x ∈ A ] B
#+end_src
